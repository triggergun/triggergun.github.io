# centos安装lsof

# 安装命令

yum -y install lsof
1

# 需求描述

本主机上有一个程序在运行。知道端口号是8080。通过端口号找到对应的进程id。并且要杀死这个进程id。

sudo lsof -i :8080
1

执行的结果：

[root@localhost nps]# sudo lsof -i :8080
COMMAND  PID USER   FD   TYPE DEVICE SIZE/OFF NODE NAME
nps     1072 root    9u  IPv6  29789      0t0  TCP *:webcache (LISTEN)
1
2
3

PID 是 1072

# 杀死进程id

sudo kill -9 <PID>
1
