			let keywords = this.keywords;
			console.log(keywords);
			if (keywords.trim()) {
				// 输入有值。跳转页面。
				this.SET_HISTORY_KEYWORDS({ keywords: keywords });
                
                // 跳转代码。
				uni.navigateTo({
				    url:"/pages/search/index?keywords="+encodeURIComponent(keywords)
				})
				
			} else {
				uni.showToast({
					title: '请输入关键词',
					icon: 'none',
					duration: 2000
				});
			}
		},
