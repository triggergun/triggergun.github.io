# 03)实现拍照和重拍功能

创建clickBtn() 方法和 afresh() 方法，实现拍照按钮和重拍的点击事件。

	clickBtn:function(){
				let that=this;
				if(that.btnText=="拍照"){
					let ctx=uni.createCameraContext();
					ctx.takePhoto({
						quality:"high",// 拍照质量
						success:function(resp){
							console.log("获取拍照图片路径:"+resp.tempImagePath)
                            
                            // 赋值给模型层变量。
							that.photoPath=resp.tempImagePath
							that.showCamera=false
							that.showImage=true
							that.btnText="签到"
						}
					})
				}
				else{
                    // 点击签到按钮要执行的代码
					uni.showLoading({
						title:"签到中请稍后"
					})
				
                    
                    //  签到要做
				
				}
			},
                
        // 重新拍照
			afresh:function(){
				let that=this;
				that.showCamera=true;
				that.showImage=false;// 图片隐藏起来图片
                // 页面的文字改为，拍照。
				that.btnText="拍照"
			}
		}
