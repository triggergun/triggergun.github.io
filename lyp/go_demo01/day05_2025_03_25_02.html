                //发送文本消息
                sendtxtmsg: function (txt) {
                    //{id:1,userid:2,dstid:3,cmd:10,media:1,content:"hello"}
                    var msg = this.createmsgcontext();
                    msg.media = 1;
                    msg.content = txt;
                    // 显示消息
                    // userInfo()怎么获取用的信息？
                    this.showmsg(userInfo(), msg);
                    this.webSocket.send(JSON.stringify(msg))
