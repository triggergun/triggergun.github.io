from flask import Flask, request, jsonify

# 定义全局变量
port = 11111

# 构建接口返回结果
def build_api_result(code, message, data,file_name,ocr_identify_time):
    result = {
        "code": code,
        "message": message,
        "data": data,
        "FileName": file_name,
        "ocrIdentifyTime": ocr_identify_time
    }

    # 对象json化
    return jsonify(result)
